   266→/* ─── Design Preview Strip ─── */
   267→function DesignPreviewStrip({ tokens, darkMode }: { tokens: Design["tokens"]; darkMode: boolean }) {
   268→  const bg = darkMode ? tokens.bgDark : tokens.bg;
   269→  const surface = darkMode ? tokens.surfaceDark : tokens.surface;
   270→  const primary = darkMode ? tokens.primaryDark : tokens.primary;
   271→  const text = darkMode ? tokens.textDark : tokens.text;
   272→  const muted = darkMode ? tokens.mutedDark : tokens.muted;
   273→  const border = darkMode ? tokens.borderDark : tokens.border;
   274→  const font = tokens.font.split(",")[0].replace(/'/g, "").trim();
   275→
   276→  return (
   277→    <div
   278→      className="design-preview-strip rounded-t-xl p-4 relative"
   279→      style={{
   280→        background: bg,
   281→        fontFamily: font,
   282→        transition: "transform 0.2s ease",
   283→      }}
   284→    >
   285→      {/* Mini layout preview */}
   286→      <div className="flex items-center gap-2 mb-3">
   287→        <div className="w-2.5 h-2.5 rounded-full" style={{ background: primary }} />
   288→        <div className="h-3 rounded-sm" style={{ background: text, width: "60px", opacity: 0.7 }} />
   289→        <div className="flex-1" />
   290→        <div className="h-2 rounded-full" style={{ background: muted, width: "40px" }} />
   291→      </div>
   292→
   293→      {/* Heading line */}
   294→      <div className="mb-3">
   295→        <div
   296→          className="text-lg font-bold leading-tight"
   297→          style={{ color: text, fontFamily: tokens.headingFont.split(",")[0].replace(/'/g, "") }}
   298→        >
   299→          The Quick Brown Fox
   300→        </div>
   301→      </div>
   302→
   303→      {/* Sample UI elements */}
   304→      <div className="flex items-center gap-2 mb-3">
   305→        <div
   306→          className="px-3 py-1.5 rounded-md text-xs font-medium"
   307→          style={{
   308→            background: primary,
   309→            color: bg === "#000000" || bg === "#0A0A0A" || bg === "#050505" || bg === "#121212" || bg === "#0C0C0C" || bg === "#0D0D0D" || bg === "#0B0E11" || bg === "#0A0A1A" || bg === "#0B0C0E" || bg === "#000000" ? "#FFFFFF" : (bg === "#FFFFFF" || bg === "#FEFEFE" || bg === "#FAFAFA" || bg === "#FDF8F4" || bg === "#FAF9F7" || bg === "#FAF8F5" ? (isLightColor(primary) ? "#111111" : "#FFFFFF") : "#FFFFFF"),
   310→            borderRadius: tokens.radius,
   311→          }}
   312→        >
   313→          Primary Action
   314→        </div>
   315→        <div
   316→          className="px-3 py-1.5 rounded-md text-xs font-medium"
   317→          style={{
   318→            background: "transparent",
   319→            color: text,
   320→            border: `1px solid ${border}`,
   321→            borderRadius: tokens.radius,
   322→          }}
   323→        >
   324→          Secondary
   325→        </div>
   326→      </div>
   327→
   328→      {/* Color dots */}
   329→      <div className="flex items-center gap-1.5">
   330→        {[primary, surface, muted, border].map((color, i) => (
   331→          <div
   332→            key={i}
   333→            className="w-4 h-4 rounded-full border"
   334→            style={{ background: color, borderColor: border }}
   335→          />
   336→        ))}
   337→      </div>
   338→    </div>
   339→  );
   340→}

   342→function isLightColor(hex: string): boolean {
   343→  const c = hex.replace("#", "");
   344→  const r = parseInt(c.substr(0, 2), 16);
   345→  const g = parseInt(c.substr(2, 2), 16);
   346→  const b = parseInt(c.substr(4, 2), 16);
   347→  return (r * 299 + g * 587 + b * 114) / 1000 > 150;
   348→}

   657→export default function HomePage() {
   658→  const [data, setData] = useState<DesignsData | null>(null);
   659→  const [darkMode, setDarkMode] = useState(false);
   660→  const [searchQuery, setSearchQuery] = useState("");
   661→  const [selectedCategory, setSelectedCategory] = useState("All");
   662→  const [selectedStyle, setSelectedStyle] = useState<string | null>(null);
   663→  const [selectedDesign, setSelectedDesign] = useState<Design | null>(null);
   664→  const [showSubmitModal, setShowSubmitModal] = useState(false);
   665→
   666→  useEffect(() => {
   667→    fetch("/data/designs.json")
   668→      .then((r) => r.json())
   669→      .then(setData);
   670→  }, []);
   671→
   672→  useEffect(() => {
   673→    document.documentElement.classList.toggle("dark", darkMode);
   674→  }, [darkMode]);
   675→
   676→  const filteredDesigns = useMemo(() => {
   677→    if (!data) return [];
   678→    return data.designs.filter((d) => {
   679→      if (selectedCategory !== "All" && d.category !== selectedCategory) return false;
   680→      if (selectedStyle && d.style !== selectedStyle) return false;
   681→      if (searchQuery) {
   682→        const q = searchQuery.toLowerCase();
   683→        return (
   684→          d.name.toLowerCase().includes(q) ||
   685→          d.vibe.toLowerCase().includes(q) ||
   686→          d.category.toLowerCase().includes(q) ||
   687→          d.style.toLowerCase().includes(q)
   688→        );
   689→      }
   690→      return true;
   691→    });
   692→  }, [data, selectedCategory, selectedStyle, searchQuery]);
   693→
   694→  const handleExplore = useCallback(() => {
   695→    const grid = document.getElementById("design-grid");
   696→    if (grid) grid.scrollIntoView({ behavior: "smooth" });
   697→  }, []);
   698→
   699→  const handleRandom = useCallback(() => {
   700→    if (!data) return;
   701→    const random = data.designs[Math.floor(Math.random() * data.designs.length)];
   702→    setSelectedDesign(random);
   703→  }, [data]);
   704→
   705→  const categoryList = useMemo(() => (data ? Object.keys(data.meta.categories) : []), [data]);
   706→
   707→  if (!data) {
   708→    return (
   709→      <div className="min-h-screen flex items-center justify-center">
   710→        <div className="text-muted-foreground text-sm">Loading designs...</div>
   711→      </div>
   712→    );
   713→  }
   714→
   715→  return (
   716→    <div className="min-h-screen flex flex-col">
   717→      <TopNav
   718→        darkMode={darkMode}
   719→        setDarkMode={setDarkMode}
   720→        searchQuery={searchQuery}
   721→        setSearchQuery={setSearchQuery}
   722→        setShowDetail={() => {}}
   723→        setShowSubmitModal={setShowSubmitModal}
   724→      />
   725→
   726→      <div className="flex flex-1">
   727→        <SideNav
   728→          categories={data.meta.categories}
   729→          selectedCategory={selectedCategory}
   730→          setSelectedCategory={setSelectedCategory}
   731→          selectedStyle={selectedStyle}
   732→          setSelectedStyle={setSelectedStyle}
   733→        />
   734→
   735→        <main className="flex-1 min-w-0">
   736→          <div className="astryx-container py-6">
   737→            <HeroCard onExplore={handleExplore} onRandom={handleRandom} />
   738→            <StatsRow total={data.meta.total} categories={data.meta.categories} designs={data.designs} />
   739→            <FilterToolbar
   740→              categories={categoryList}
   741→              activeCategory={selectedCategory}
   742→              setActiveCategory={setSelectedCategory}
   743→            />
   744→
   745→            {/* Results count */}
   746→            <div className="flex items-center justify-between mb-4">
   747→              <p className="text-sm text-muted-foreground">
   748→                Showing <span className="font-medium text-foreground">{filteredDesigns.length}</span> design{filteredDesigns.length !== 1 ? "s" : ""}
   749→              </p>
   750→            </div>
   751→
   752→            {/* Grid */}
   753→            <div id="design-grid" className="grid grid-cols-1 md:grid-cols-2 xl:grid-cols-3 gap-4">
   754→              {filteredDesigns.map((design) => (
   755→                <DesignCard
   756→                  key={design.id}
   757→                  design={design}
   758→                  darkMode={darkMode}
   759→                  onSelect={setSelectedDesign}
   760→                />
   761→              ))}
   762→            </div>
   763→
   764→            {filteredDesigns.length === 0 && (
   765→              <div className="text-center py-16">
   766→                <p className="text-muted-foreground text-sm">No designs match your filters.</p>
   767→                <button
   768→                  onClick={() => {
   769→                    setSelectedCategory("All");
   770→                    setSelectedStyle(null);
   771→                    setSearchQuery("");
   772→                  }}
   773→                  className="astryx-button mt-3 text-sm"
   774→                >
   775→                  Clear all filters
   776→                </button>
   777→              </div>
   778→            )}
   779→          </div>
   780→
   781→          <Footer />
   782→        </main>
   783→      </div>
   784→
   785→      {/* Detail Panel */}
   786→      {selectedDesign && (
   787→        <DesignDetailPanel
   788→          design={selectedDesign}
   789→          darkMode={darkMode}
   790→          onClose={() => setSelectedDesign(null)}
   791→        />
   792→      )}
   793→
   794→      {/* Submit Modal */}
   795→      {showSubmitModal && <SubmitModal onClose={() => setShowSubmitModal(false)} />}
   796→    </div>
   797→  );
